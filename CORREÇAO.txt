using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using static System.Console;

namespace ConsoleApp1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Write("Numeros primos 1 a 100:");
            for (int num = 0; num <= 100; num++)
            {
                bool primo = true;
                for (int div = 2;div <= Math.Sqrt(num);div++) 
                {
                    if (num % div == 0)
                    {
                        primo = false;
                        break;
                    }
                }
                if (primo) 
                {
                    WriteLine(num + " ");
                }
            }
            ReadKey();
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using static System.Console;

namespace ConsoleApp1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int soma = 0, count = 0, num;
            WriteLine("Digite numeros inteiros(0 para ter a conclusão final da media):");
            while ((num = int.Parse(ReadLine())) != 0)
            {
                soma += num;
                count++;
            }
            if (count > 0 )
            {
                WriteLine($"Media: {(double)soma / count}");
            }
            else
            {
                WriteLine("Nenhum numero foi digitado. ");
            }           
            ReadKey();
        }
    }
}
